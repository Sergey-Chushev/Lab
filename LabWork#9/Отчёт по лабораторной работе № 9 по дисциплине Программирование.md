#ifndef TST_TESTDIVISIONBYZERO_H

#include <ILatch.h>
#include <IKeypad.h>

class MockILatch: public ILatch{
public:
    MOCK_METHOD(
            bool,
            isActive,
            (),
            (override));
    MOCK_METHOD(
            DoorStatus,
            open,
            (),
            (override)
            );
    MOCK_METHOD(
            DoorStatus,
            close,
            (),
            (override)
            );
    MOCK_METHOD(
           DoorStatus,
           getDoorStatus,
           (),
           (override)
           );
};

class MockIKeypad: public IKeypad{
public:
   MOCK_METHOD(
           bool,

   MOCK_METHOD (

Ошибка

```
DoorStatus LockController::lockDoor(){
    return latch->open();
}
```

Исправление

```
DoorStatus LockController::lockDoor(){
    return latch->close();
}
```

Ошибка возникла из-за отсутствия вызова метода Close.
